== should format scss constructs ==
$primary:   #333;
.btn {
  color: $primary;
  &:hover{color:lighten($primary,10%)}
  // line comment
  .icon-#{$name}{width:10px}
}

[expect]
$primary: #333;
.btn {
  color: $primary;
  &:hover {
    color: lighten($primary, 10%);
  }
  // line comment
  .icon-#{$name} {
    width: 10px;
  }
}

== should format mixins ==
@mixin foo($a,$b: 10px){margin:$a $b}
.box{@include foo(1px);}

[expect]
@mixin foo($a, $b: 10px) {
  margin: $a $b;
}
.box {
  @include foo(1px);
}
