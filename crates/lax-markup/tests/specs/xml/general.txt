== should format xml structurally ==
<?xml version="1.0" encoding="UTF-8"?>
<root>
<item id="1">
<child/>
</item>
</root>

[expect]
<?xml version="1.0" encoding="UTF-8"?>
<root>
  <item id="1">
    <child />
  </item>
</root>

</note>

<note>
  <body>Dear <name>someone</name>, hello</body>
</note>
